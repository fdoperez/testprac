# To run cmake successfully on windows : cmake -G MinGW Makefiles ..
# To run make successfully on windows  : mingw32-make

cmake_minimum_required(VERSION 3.10)

# Set the project name
project(pr1_IAA)

# Set the project minimum standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Establecer el directorio para los ejecutables
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/bin)

# Asegurar que la carpeta exista
file(MAKE_DIRECTORY ${CMAKE_SOURCE_DIR}/bin)

# Buscar los archivos fuente
set(SOURCES
  # Añade las fuentes
  ${CMAKE_SOURCE_DIR}/src/main.cc
  ${CMAKE_SOURCE_DIR}/src/binary_distribution.cc
  ${CMAKE_SOURCE_DIR}/src/interface.cc
  ${CMAKE_SOURCE_DIR}/src/chronometer.cc
)

# Crear el ejecutable
add_executable(pr1_IAA ${SOURCES})

# Incluir el directorio include para que CMake lo sepa
target_include_directories(pr1_IAA PRIVATE ${CMAKE_SOURCE_DIR}/include)
